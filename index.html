// Hàm xử lý gom dữ liệu 12 tháng theo CHU KỲ và vẽ biểu đồ đè lồng nhau (Đã sửa lỗi cache) function capNhatBieuDoBaoCaoNam(docs) { const ctx = document.getElementById('bieuDoNganSach'); if (!ctx) return; // Lấy năm hiện tại dựa trên ô chọn tháng đang xem const namDangXem = window.thoiGianDangXem.getFullYear(); // Đổi chữ tiêu đề theo năm đang hiển thị const txtTieuDeBc = document.getElementById('tieu-de-bieu-do'); if (txtTieuDeBc) txtTieuDeBc.innerText = `Biểu đồ Ngân sách & Chi tiêu năm ${namDangXem}`; // Mảng lưu tổng chi tiêu thực tế gốc của 12 tháng chu kỳ const mangDaChiThucTe = Array(12).fill(0); // Duyệt qua từng giao dịch để phân loại vào THÁNG CHU KỲ docs.forEach(docItem => { const data = docItem.data(); if (data.loai === 'chi' && data.thoiGian) { const thoiGianMoc = parseChuoiThoiGian(data.thoiGian); for (let m = 0; m < 12; m++) { const ngayDaiDienThang = new Date(namDangXem, m, 1); const chuKyMoc = layRangeChuKy(ngayDaiDienThang); if (thoiGianMoc >= chuKyMoc.tuNgay && thoiGianMoc <= chuKyMoc.denNgay) { mangDaChiThucTe[m] += parseFloat(data.soTien) || 0; break; } } } }); // Tạo các mảng động cấu trúc biểu đồ lồng nhau const mangMax = Array(12).fill(0); const mangMin = Array(12).fill(0); const mangMauMax = Array(12).fill(''); const mangMauMin = Array(12).fill(''); const mangLoaiMax = Array(12).fill(''); const mangLoaiMin = Array(12).fill(''); // So sánh từng tháng để quyết định thứ tự trước/sau và màu sắc for (let m = 0; m < 12; m++) { const chuoiThangNam = `${namDangXem}-${String(m + 1).padStart(2, '0')}`; const hanMuc = window.boSuoTapHanMuc[chuoiThangNam] || 0; const daChi = mangDaChiThucTe[m]; if (hanMuc >= daChi) { // Trường hợp AN TOÀN: Ngân sách lớn làm nền (Xanh), Chi tiêu nhỏ nằm trước (Đỏ) mangMax[m] = hanMuc; mangMin[m] = daChi; mangMauMax[m] = '#3498db'; mangMauMin[m] = '#e74c3c'; mangLoaiMax[m] = 'Ngân sách'; mangLoaiMin[m] = 'Thực tế chi'; } else { // Trường hợp VƯỢT MỨC: Chi tiêu vượt vọt lên làm nền (Đỏ), Ngân sách thu gọn ở trước (Xanh) mangMax[m] = daChi; mangMin[m] = hanMuc; mangMauMax[m] = '#e74c3c'; mangMauMin[m] = '#3498db'; mangLoaiMax[m] = 'Thực tế chi'; mangLoaiMin[m] = 'Ngân sách'; } } // 🔥 BƯỚC QUAN TRỌNG NHẤT: Nếu đã có biểu đồ cũ, hủy hoàn toàn để xóa sạch cấu hình "tách cột" cũ if (bieuDoInstance) { bieuDoInstance.destroy(); } // Khởi tạo mới hoàn toàn biểu đồ với cấu hình ép đè lồng bắt buộc bieuDoInstance = new Chart(ctx, { type: 'bar', data: { labels: ['T1', 'T2', 'T3', 'T4', 'T5', 'T6', 'T7', 'T8', 'T9', 'T10', 'T11', 'T12'], datasets: [ { label: 'Thanh Sau', data: mangMax, backgroundColor: mangMauMax, barPercentage: 0.85, categoryPercentage: 0.8, // Giữ tỷ lệ khoảng cách đồng bộ giữa các tháng borderRadius: 4, grouped: false // 🔑 Ép buộc thanh này không chia tách không gian cột }, { label: 'Thanh Trước', data: mangMin, backgroundColor: mangMauMin, barPercentage: 0.55, // Co nhỏ thanh trước lại để lộ viền thanh sau ra ngoài categoryPercentage: 0.8, borderRadius: 4, grouped: false // 🔑 Ép buộc thanh này nhảy vào đứng chung tâm với thanh sau } ] }, options: { responsive: true, maintainAspectRatio: false, plugins: { legend: { display: false }, tooltip: { callbacks: { label: function(context) { const m = context.dataIndex; const val = context.raw; const loai = context.datasetIndex === 0 ? mangLoaiMax[m] : mangLoaiMin[m]; return loai + ': ' + val.toLocaleString('vi-VN') + ' đ'; } } } }, scales: { x: { stacked: false // Đảm bảo các cột không chồng nối đuôi nhau lên trời mà là đè lấp lên nhau }, y: { beginAtZero: true, ticks: { callback: function(value) { return (value / 1000000).toFixed(1) + 'M'; } } } } } }); }
